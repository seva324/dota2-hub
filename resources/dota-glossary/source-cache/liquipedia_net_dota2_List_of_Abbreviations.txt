( function() {

	// Sampling configuration: 0.1% = 1/1000 (modulo 0)
	const SAMPLING_MODULO = 1000;
	const SAMPLING_THRESHOLD = 0; // Change the threshold to adjust the sampling rate

	function lpAdsCheckSampling() {
		const CACHE_KEY = 'lpads_amplitude_sampling';
		const USER_ID_KEY = 'lpads_amplitude_userid';

		const userId = '';
		const cachedUserId = localStorage.getItem( USER_ID_KEY );
		const cachedDecision = localStorage.getItem( CACHE_KEY );

		if ( userId && userId !== '' ) {
			if ( cachedUserId === userId && cachedDecision !== null ) {
				const result = cachedDecision === 'true';
				return result;
			}

			// userId is already hashed server-side (HMAC-SHA256), parse as hex number
			const userNumeric = parseInt( userId.slice( 0, 8 ), 16 );
			const userModulo = userNumeric % SAMPLING_MODULO;
			const userInSample = userModulo === SAMPLING_THRESHOLD;

			localStorage.setItem( USER_ID_KEY, userId );
			localStorage.setItem( CACHE_KEY, String( userInSample ) );
			return userInSample;
		}

		if ( cachedUserId === '' && cachedDecision !== null ) {
			const result = cachedDecision === 'true';
			return result;
		}

		const apiKey = 'dd56ce9695b924d3f30715c56c6cc953';
		let deviceId = localStorage.getItem( `AMP_UNSENT_${ apiKey }` );
		if ( !deviceId ) {
			deviceId = 'temp_' + Date.now() + '_' + Math.random().toString( 36 ).slice( 2, 15 );
		}

		// Convert deviceId string to numeric value using character codes
		let deviceNumeric = 0;
		for ( let i = 0; i < deviceId.length; i++ ) {
			deviceNumeric += deviceId.charCodeAt( i );
		}
		const deviceModulo = deviceNumeric % SAMPLING_MODULO;
		const deviceInSample = deviceModulo === SAMPLING_THRESHOLD;

		localStorage.setItem( USER_ID_KEY, '' );
		localStorage.setItem( CACHE_KEY, String( deviceInSample ) );
		return deviceInSample;
	}

	const lpAdsSamplingPlugin = {
		name: 'liquipedia-ads-sampling-plugin',
		type: 'enrichment',
		inSample: false,
		eventCount: 0,
		droppedCount: 0,

		setup: function() {
			this.inSample = lpAdsCheckSampling();
			return undefined;
		},

		execute: function( event ) {
			this.eventCount++;
			if ( this.inSample ) {
				return event;
			} else {
				this.droppedCount++;
				return null;
			}
		}
	};

	function lpAdsLoadAmplitudeSDK() {
		return new Promise( ( resolve, reject ) => {
			if ( typeof amplitude !== 'undefined' ) {
				resolve();
				return;
			}

			const existingScript = document.querySelector( 'script[src*="amplitude.com"]' );
			if ( existingScript ) {
				existingScript.addEventListener( 'load', () => {
					resolve();
				} );
				existingScript.addEventListener( 'error', ( error ) => {
					reject( error );
				} );
			} else {
				reject( new Error( 'Amplitude SDK script tag not found' ) );
			}
		} );
	}

	async function lpAdsInitAmplitude() {
		const apiKey = 'dd56ce9695b924d3f30715c56c6cc953';
